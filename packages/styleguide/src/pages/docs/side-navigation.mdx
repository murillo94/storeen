---
title: 'Side Navigation'
component: 'SideNavigation'
description: 'Useful for displaying one panel of content with link at a time.'
---

```js live
<SideNavigation>
  <SideNavigationGroup>
    <SideNavigationItem isActive>
      <SideNavigationItemLink href="/" isExternal>
        im side navigation item 1
      </SideNavigationItemLink>
    </SideNavigationItem>
    <SideNavigationItem>
      <SideNavigationItemLink href="/">
        im side navigation item 2
      </SideNavigationItemLink>
    </SideNavigationItem>
  </SideNavigationGroup>
</SideNavigation>
```
